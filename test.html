< describe 'posts#create action' do it 'saves the post in the database' do user = FactoryGirl.create(:user) sign_in user post :create, params: { post: { message: 'Hi', picture: fixture_file_upload('/picture.png', 'image/png') } } expect(response).to redirect_to root_path post = Post.last expect(post.message).to eq('Hi') expect(post.user).to eq(user) end it 'deals with errors correctly' do user = FactoryGirl.create(:user) sign_in user post :create, params: { post: { message: '' } } expect(response).to have_http_status(:unprocessable_entity) expect(Post.count).to eq 0 end end describe 'posts#update action' do it 'allows users to update their posts' do post = FactoryGirl.create(:post, message: 'First Value') sign_in post.user patch :update, params: { id: post.id, post: { message: 'Changed Value' } } expect(response).to redirect_to root_path post.reload expect(post.message).to eq 'Changed Value' end it 'will not let unauthorized users update a post' do post = FactoryGirl.create(:post) patch :update, params: { id: post.id, post: { message: 'Not my post!' } } expect(response).to redirect_to new_user_session_path end end describe 'posts#edit action' do it 'should show the edit form to the correct user' do post = FactoryGirl.create(:post) sign_in post.user get :edit, params: { id: post.id } expect(response).to have_http_status(:success) end it 'will not let unauthorized users edit the post' do post = FactoryGirl.create(:post) user = FactoryGirl.create(:user) sign_in user get :edit, params: { id: post.id, post: { message: 'Not my post!' } } expect(response).to have_http_status(:forbidden) end it 'shows a 404 error if the post to edit is not found' do user = FactoryGirl.create(:user) sign_in user get :edit, params: { id: 'Not found' } expect(response).to have_http_status(:not_found) end end
